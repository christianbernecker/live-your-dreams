Next.js Backoffice Integration

// components/PropertyAccordion.tsx
'use client';

import { useState, useEffect } from 'react';

export default function PropertyAccordion({ property }: { property: Property }) {
  const [expandedSections, setExpandedSections] = useState(['basic']);

  useEffect(() => {
    import('@/lib/lyd-design-system');
  }, []);

  const toggleSection = (section: string) => {
    setExpandedSections(prev => 
      prev.includes(section) 
        ? prev.filter(s => s !== section)
        : [...prev, section]
    );
  };

  return (
    <div className="property-details">
      <div className="luxury-accordion-group">
        <div className={`accordion-item ${expandedSections.includes('basic') ? 'expanded' : ''}`}>
          <button 
            className="accordion-trigger"
            onClick={() => toggleSection('basic')}
          >
            <span>Basic Information</span>
            <svg className="accordion-icon" viewBox="0 0 24 24">
              <polyline points="6,9 12,15 18,9"></polyline>
            </svg>
          </button>
          <div className="accordion-content">
            <div className="property-info">
              <p><strong>Type:</strong> {property.type}</p>
              <p><strong>Size:</strong> {property.area} m²</p>
              <p><strong>Rooms:</strong> {property.rooms}</p>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}

// components/PropertyAccordion.tsx
'use client';

import { useState } from 'react';

interface AccordionItem {
  id: string;
  title: string;
  content: React.ReactNode;
}

export default function PropertyAccordion({ items }: { items: AccordionItem[] }) {
  const [openItems, setOpenItems] = useState<string[]>([]);

  const toggleItem = (id: string) => {
    setOpenItems(prev => 
      prev.includes(id) 
        ? prev.filter(item => item !== id)
        : [...prev, id]
    );
  };

  return (
    <div className="accordion-container">
      {items.map(item => (
        <div key={item.id} className="accordion-item">
          <button 
            className={`accordion-header ${openItems.includes(item.id) ? 'active' : ''}`}
            onClick={() => toggleItem(item.id)}
          >
            {item.title}
            <span className="accordion-icon">
              {openItems.includes(item.id) ? '‚ñ≤' : '‚ñº'}
            </span>
          </button>
          
          <div className={`accordion-content ${openItems.includes(item.id) ? 'active' : ''}`}>
            <div className="accordion-body">
              {item.content}
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}

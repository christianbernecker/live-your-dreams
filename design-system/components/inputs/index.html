Properties

Property Type Default Description
variant string default Input style variant: default, search, currency, area
label string - Input label text
placeholder string - Placeholder text
helper string - Helper text below input
error string - Error message (shows error state)
required boolean false Shows required indicator
disabled boolean false Disables the input
multiline boolean false Renders as textarea

Next.js Usage

// app/properties/new/page.tsx
'use client';

import { useState, useEffect } from 'react';

export default function NewPropertyPage() {
  const [formData, setFormData] = useState({});
  
  useEffect(() => {
    import('@/lib/lyd-design-system');
  }, []);

  const handleInputChange = (e) => {
    setFormData(prev => ({
      ...prev,
      [e.target.name]: e.detail.value
    }));
  };

  return (
    <div className="p-8">
      <lyd-input 
        name="title"
        label="Property Title"
        placeholder="Enter property title..."
        required
        onInput={handleInputChange}
      />
      
      <lyd-input 
        name="price"
        variant="currency"
        label="Purchase Price"
        placeholder="750,000"
        onInput={handleInputChange}
      />
      
      <lyd-input 
        name="area"
        variant="area"
        label="Living Space (m²)"
        placeholder="120"
        onInput={handleInputChange}
      />
    </div>
  );
}

// Next.js Integration
'use client';

import { useEffect, useState } from 'react';

export default function PropertyForm() {
  const [formData, setFormData] = useState({});
  const [errors, setErrors] = useState({});

  useEffect(() => {
    // LYD Design System Web Components laden
    import('@/lib/lyd-design-system');
  }, []);

  const handleInputChange = (e) => {
    const { name } = e.target;
    const { value } = e.detail;
    
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
  };

  const handleValidation = (e) => {
    const { name } = e.target;
    const { isValid, errors } = e.detail;
    
    if (!isValid) {
      setErrors(prev => ({
        ...prev,
        [name]: errors[0]
      }));
    } else {
      setErrors(prev => {
        const newErrors = { ...prev };
        delete newErrors[name];
        return newErrors;
      });
    }
  };

  return (
    <form className="space-y-6">
      {/* Standard Text Input */}
      <lyd-input-text
        name="title"
        label="Immobilien-Titel"
        placeholder="Moderne Wohnung in München"
        required
        onInput={handleInputChange}
        onValidation={handleValidation}
      />
      
      {/* Email Input mit Domain-Validierung */}
      <lyd-input-email
        name="contact-email"
        label="Kontakt E-Mail"
        placeholder="kontakt@immobilien.de"
        icon-left="mail"
        validate-domain="immobilien.de,remax.de"
        onInput={handleInputChange}
      />
      
      {/* Immobilien-spezifische Price Input */}
      <lyd-input-price
        name="price"
        label="Kaufpreis"
        price-type="sale"
        show-per-sqm
        area-reference="living-area"
        onInput={handleInputChange}
      />
      
      {/* Area Input mit Einheiten-Umrechnung */}
      <lyd-input-area
        name="living-area"
        label="Wohnfläche"
        area-type="living"
        unit="m²"
        show-unit-converter
        onInput={handleInputChange}
      />
      
      {/* Composite Address Input */}
      <lyd-input-address
        name="property-address"
        validate-address
        show-map
        onAddressChange={(e) => setFormData(prev => ({
          ...prev,
          address: e.detail
        }))}
      />
      
      {/* Search Input mit Autocomplete */}
      <lyd-input-search
        name="property-search"
        label="Immobilie suchen"
        placeholder="München, Wohnung, 3 Zimmer..."
        debounce={300}
        min-chars={2}
        show-suggestions
        onSearch={(e) => performSearch(e.detail.query)}
        onSelect={(e) => selectProperty(e.detail)}
      />
    </form>
  );
}

// Real Estate Validation Rules
const validatePropertyPrice = (price, type) => {
  const numPrice = parseFloat(price);
  
  if (type === 'sale' && numPrice < 10000) {
    return 'Kaufpreis muss mindestens €10.000 betragen';
  }
  
  if (type === 'rent' && numPrice < 100) {
    return 'Miete muss mindestens €100 betragen';
  }
  
  return null;
};

Barrierefreiheit & Standards

WCAG 2.1 AA Konformität

Alle Input-Komponenten erfüllen die internationalen Web Content Accessibility Guidelines und unterstützen assistive Technologien vollständig.

Tastaturnavigation

  • Tab - Navigation zwischen Input-Feldern
  • Shift + Tab - Rückwärts-Navigation
  • Enter - Formular absenden (bei Submit-Button Fokus)
  • Esc - Autocomplete-Vorschläge schließen
  • Pfeil-Tasten - Navigation in Dropdown-Listen und Datepicker

Screen Reader Unterstützung

  • Alle Labels sind semantisch mit ihren Input-Feldern verknüpft
  • Error-Messages werden mit role="alert" sofort angekündigt
  • Pflichtfelder werden mit aria-required="true" markiert
  • Validierungszustände werden mit aria-invalid kommuniziert
  • Helper-Text wird mit aria-describedby verknüpft
  • Autocomplete-Listen haben role="listbox" und aria-expanded
  • Ladezustände werden mit aria-busy angezeigt

Farbkontrast & Visuelle Zugänglichkeit

  • Text zu Hintergrund: Mindestens 4.5:1 Kontrastverhältnis
  • Fokus-Indikatoren: 3:1 Kontrastverhältnis zu angrenzenden Farben
  • Error States: Nicht nur durch Farbe kommuniziert, sondern auch durch Icons und Text
  • Placeholder Text: Ausreichender Kontrast ohne Abhängigkeit für wichtige Informationen
  • Support für Windows High Contrast Mode
  • Skalierung bis 200% ohne Funktionalitätsverlust

Form Association API

  • Vollständige Integration in HTML5 Forms
  • Native Browser-Validierung unterstützt
  • Automatische Form-Submission mit korrekten Namen/Werten
  • Reset-Funktionalität für alle Input-Zustände
  • Custom Validity API für erweiterte Validierung

Mobile & Touch Accessibility

  • Minimale Touch-Target-Größe von 44px × 44px
  • Optimierte Bildschirmtastatur für Input-Typen (email, tel, number)
  • Zoom-freundlich ohne horizontales Scrollen
  • Voice-Over und TalkBack Unterstützung
  • Haptic Feedback bei Validierungsfehlern (unterstützte Geräte)

Internationalisierung

  • RTL (Right-to-Left) Layout-Support
  • Lokalisierte Datums- und Zahlenformate
  • Kulturspezifische Validierungsregeln (Telefonnummern, Postleitzahlen)
  • Übersetzbare Error-Messages und Labels
  • Currency- und Unit-Formatting basierend auf Locale

API Reference

lyd-input-text

Standard text input component with validation and error states.

Properties

Property Type Default Description
label string - Input label text
placeholder string - Placeholder text
required boolean false Mark field as required
disabled boolean false Disable input
size 'small' | 'medium' | 'large' 'medium' Input size variant
variant 'default' | 'outline' | 'filled' 'default' Input visual variant

Events

  • lyd-change - Fired when input value changes
  • lyd-focus - Fired when input receives focus
  • lyd-blur - Fired when input loses focus

Next.js Integration

// app/properties/create/page.tsx
'use client';

import { useEffect, useState } from 'react';

export default function CreatePropertyPage() {
  const [formData, setFormData] = useState({});

  useEffect(() => {
    import('@/lib/lyd-design-system');
  }, []);

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('Form data:', formData);
  };

  return (
    <form onSubmit={handleSubmit}>
      <lyd-input-text 
        label="Property Title" 
        required
        onChange={(e) => setFormData({...formData, title: e.target.value})}
      />
      
      <lyd-input-price 
        label="Price" 
        currency="EUR"
        onChange={(e) => setFormData({...formData, price: e.target.value})}
      />
      
      <lyd-input-area 
        label="Area" 
        unit="m2"
        onChange={(e) => setFormData({...formData, area: e.target.value})}
      />
      
      <lyd-button type="submit" variant="primary">
        Create Property
      </lyd-button>
    </form>
  );
}

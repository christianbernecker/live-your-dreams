Next.js Backoffice Integration

// app/properties/page.tsx
'use client';

import { useEffect, useState } from 'react';

export default function PropertiesPage() {
  const [isLoading, setIsLoading] = useState(false);
  const [properties, setProperties] = useState([]);

  useEffect(() => {
    // Load LYD Design System
    import('@/lib/lyd-design-system');
  }, []);

  const handleCreateProperty = async () => {
    setIsLoading(true);
    try {
      const response = await fetch('/api/properties', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ title: 'New Property' })
      });
      const newProperty = await response.json();
      setProperties([...properties, newProperty]);
    } finally {
      setIsLoading(false);
    }
  };

  const handleDownloadExpose = async (propertyId: string) => {
    const response = await fetch(`/api/properties/${propertyId}/expose`);
    const blob = await response.blob();
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `property-${propertyId}-expose.pdf`;
    a.click();
  };

  return (
    <div className="min-h-screen bg-gray-50 py-8">
      <div className="max-w-7xl mx-auto px-4">
        
        {/* Header Actions */}
        <div className="flex justify-between items-center mb-8">
          <h1 className="text-3xl font-bold">Properties</h1>
          
          <div className="flex gap-4">
            <lyd-button 
              variant="outline" 
              icon="upload"
              onClick={() => importProperties()}
            >
              Import Properties
            </lyd-button>
            
            <lyd-button 
              variant="primary" 
              size="large"
              icon="plus"
              loading={isLoading}
              onClick={handleCreateProperty}
            >
              Add New Property
            </lyd-button>
          </div>
        </div>

        {/* Property Cards */}
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {properties.map((property) => (
            <div key={property.id} className="bg-white rounded-xl shadow-sm border">
              <div className="p-6">
                <h3 className="font-semibold mb-2">{property.title}</h3>
                <p className="text-gray-600 mb-4">{property.description}</p>
                
                {/* Property Actions */}
                <div className="flex gap-2">
                  <lyd-button 
                    variant="primary" 
                    icon="eye" 
                    size="small"
                    onClick={() => viewProperty(property.id)}
                  >
                    View
                  </lyd-button>
                  
                  <lyd-button 
                    variant="secondary" 
                    icon="edit" 
                    size="small"
                    onClick={() => editProperty(property.id)}
                  >
                    Edit
                  </lyd-button>
                  
                  <lyd-download-button
                    variant="outline"
                    size="small"
                    file-name="Property Exposé"
                    file-size="2.5 MB"
                    onClick={() => handleDownloadExpose(property.id)}
                  />
                </div>
              </div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

// API Route Example: app/api/properties/route.ts
import { NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const property = await prisma.property.create({
      data: {
        title: body.title,
        description: body.description,
        price: body.price,
        area: body.area,
        rooms: body.rooms
      }
    });
    
    return NextResponse.json(property);
  } catch (error) {
    return NextResponse.json(
      { error: 'Failed to create property' }, 
      { status: 500 }
    );
  }
}

// Hook für Button Events
import { useEffect, useRef } from 'react';

export const useLydButton = (
  onButtonClick?: (detail: any) => void,
  onDownloadComplete?: (detail: any) => void
) => {
  const ref = useRef<HTMLDivElement>(null);
  
  useEffect(() => {
    const element = ref.current;
    if (!element) return;
    
    const handleButtonClick = (e: CustomEvent) => {
      onButtonClick?.(e.detail);
    };
    
    const handleDownloadComplete = (e: CustomEvent) => {
      onDownloadComplete?.(e.detail);
    };
    
    element.addEventListener('lyd-click', handleButtonClick as EventListener);
    element.addEventListener('download-complete', handleDownloadComplete as EventListener);
    
    return () => {
      element.removeEventListener('lyd-click', handleButtonClick as EventListener);
      element.removeEventListener('download-complete', handleDownloadComplete as EventListener);
    };
  }, [onButtonClick, onDownloadComplete]);
  
  return ref;
};

Accessibility Guidelines

WCAG 2.1 AA Compliant

All button components meet accessibility standards

Keyboard Navigation

  • Tab - Navigate between buttons
  • Space / Enter - Activate button
  • Esc - Cancel action (if applicable)

Screen Reader Support

  • All buttons have proper ARIA labels
  • Icon-only buttons include descriptive text
  • Loading states announce to screen readers
  • Disabled states are properly communicated

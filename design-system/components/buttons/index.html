Lead Actions

Call Lead Send Email Schedule Meeting

Properties

Property Type Default Description
variant string primary Button style variant: primary, secondary, outline, ghost
size string medium Button size: small, medium, large
loading boolean false Shows loading spinner and disables interaction
disabled boolean false Disables the button
fullWidth boolean false Makes button full width of container

Next.js Usage

// app/properties/page.tsx
'use client';

import { useEffect } from 'react';

export default function PropertiesPage() {
  useEffect(() => {
    import('@/lib/lyd-design-system');
  }, []);

  return (
    <div className="p-8">
      <lyd-button 
        variant="primary" 
        size="large"
        onClick={() => createProperty()}
      >
        Add New Property
      </lyd-button>
      
      <lyd-button 
        variant="secondary"
        loading={isLoading}
        disabled={!canSave}
      >
        Save Changes
      </lyd-button>
    </div>
  );
}

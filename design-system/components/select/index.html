lyd-select Properties

Property Type Default Description
variant flat | bordered | faded | underlined flat Visual style variant of the select
size sm | md | lg md Size of the select component
isMultiple boolean false Enable multiple selection with chips
isSearchable boolean false Enable search functionality
isDisabled boolean false Disable the select component
isInvalid boolean false Show error state styling
placeholder string - Placeholder text when no option is selected
label string - Label text for the select
description string - Helper text below the select
errorMessage string - Error message when isInvalid is true

Next.js Implementation

PropertySearchForm.tsx 
import { useState } from 'react';
import { useForm, Controller } from 'react-hook-form';

interface PropertySearchForm {
  propertyType: string;
  location: string;
  priceRange: string;
  amenities: string[];
}

export default function PropertySearchForm() {
  const { control, handleSubmit, watch } = useForm<PropertySearchForm>({
    defaultValues: {
      propertyType: '',
      location: '',
      priceRange: '',
      amenities: []
    }
  });
  
  const onSubmit = (data: PropertySearchForm) => {
    console.log('Search criteria:', data);
    // Handle search logic
  };
  
  return (
    <form onSubmit={handleSubmit(onSubmit)} className="property-search-form">
      <div className="form-grid">
        {/* Property Type Select */}
        <Controller
          name="propertyType"
          control={control}
          render={({ field }) => (
            <lyd-select
              {...field}
              label="Property Type"
              placeholder="Select property type"
              variant="bordered"
              size="md"
            >
              <lyd-select-item value="apartment">Apartment</lyd-select-item>
              <lyd-select-item value="house">House</lyd-select-item>
              <lyd-select-item value="villa">Villa</lyd-select-item>
              <lyd-select-item value="commercial">Commercial</lyd-select-item>
            </lyd-select>
          )}
        />
        
        {/* Searchable Location Select */}
        <Controller
          name="location"
          control={control}
          render={({ field }) => (
            <lyd-select
              {...field}
              label="Location"
              placeholder="Search location..."
              variant="bordered"
              isSearchable
              size="md"
            >
              <lyd-select-item value="munich">Munich, Germany</lyd-select-item>
              <lyd-select-item value="berlin">Berlin, Germany</lyd-select-item>
              <lyd-select-item value="hamburg">Hamburg, Germany</lyd-select-item>
            </lyd-select>
          )}
        />
        
        {/* Multiple Selection for Amenities */}
        <Controller
          name="amenities"
          control={control}
          render={({ field }) => (
            <lyd-select
              {...field}
              label="Amenities"
              placeholder="Select amenities"
              variant="bordered"
              isMultiple
              size="md"
            >
              <lyd-select-item value="pool">Swimming Pool</lyd-select-item>
              <lyd-select-item value="garden">Private Garden</lyd-select-item>
              <lyd-select-item value="garage">Garage</lyd-select-item>
              <lyd-select-item value="balcony">Balcony/Terrace</lyd-select-item>
            </lyd-select>
          )}
        />
      </div>
      
      <button type="submit" className="search-button">
        Search Properties
      </button>
    </form>
  );
}

select.types.ts 
// React TypeScript definitions for lyd-select components
declare namespace JSX {
  interface IntrinsicElements {
    'lyd-select': {
      variant?: 'flat' | 'bordered' | 'faded' | 'underlined';
      size?: 'sm' | 'md' | 'lg';
      isMultiple?: boolean;
      isSearchable?: boolean;
      isDisabled?: boolean;
      isInvalid?: boolean;
      isClearable?: boolean;
      placeholder?: string;
      label?: string;
      description?: string;
      errorMessage?: string;
      value?: string | string[];
      onChange?: (value: string | string[]) => void;
      onSelectionChange?: (keys: Set<string>) => void;
      children?: React.ReactNode;
    };
    
    'lyd-select-item': {
      value: string;
      isDisabled?: boolean;
      children?: React.ReactNode;
    };
    
    'lyd-select-section': {
      title?: string;
      children?: React.ReactNode;
    };
  }
}

// Hook for managing select state
export interface UseSelectProps {
  defaultValue?: string | string[];
  isMultiple?: boolean;
  onSelectionChange?: (value: string | string[]) => void;
}

export function useSelect(props: UseSelectProps) {
  const [selectedValue, setSelectedValue] = useState(props.defaultValue);
  const [isOpen, setIsOpen] = useState(false);
  
  const handleSelectionChange = (value: string | string[]) => {
    setSelectedValue(value);
    props.onSelectionChange?.(value);
  };
  
  return {
    selectedValue,
    isOpen,
    setIsOpen,
    handleSelectionChange
  };
}

Implementation Notes

ARIA Attributes

  • role="combobox" - Identifies the select trigger
  • aria-expanded - Indicates dropdown state
  • aria-haspopup="listbox" - Indicates dropdown content
  • aria-labelledby - Associates label with select
  • aria-describedby - Associates help text
  • aria-invalid - Indicates error state
  • aria-selected - Indicates selected options

Error Handling

  • Clear error messages with specific guidance
  • Error announcements via live regions
  • Visual and programmatic error indication
  • Focus management during error states

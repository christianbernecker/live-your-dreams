API Reference

lyd-select

Luxury dropdown select component with Porsche-level styling and animations.

Properties

Property Type Default Description
label string - Select label text
placeholder string - Placeholder text
multiple boolean false Enable multiple selection
searchable boolean false Enable search functionality
disabled boolean false Disable select
size 'small' | 'medium' | 'large' 'medium' Select size variant
variant 'default' | 'outline' | 'filled' 'default' Select visual variant

Events

  • lyd-change - Fired when selection changes
  • lyd-focus - Fired when select receives focus
  • lyd-blur - Fired when select loses focus
  • lyd-open - Fired when dropdown opens
  • lyd-close - Fired when dropdown closes

Next.js Integration

// components/PropertyFilters.tsx
'use client';

import { useState, useEffect } from 'react';

interface FilterState {
  propertyType: string;
  priceRange: string;
  bedrooms: string;
  location: string;
  features: string[];
}

export default function PropertyFilters() {
  const [filters, setFilters] = useState<FilterState>({
    propertyType: '',
    priceRange: '',
    bedrooms: '',
    location: '',
    features: []
  });

  useEffect(() => {
    import('@/lib/lyd-design-system');
  }, []);

  const handleFilterChange = (field: keyof FilterState, value: string) => {
    setFilters(prev => ({ ...prev, [field]: value }));
  };

  const handleMultiSelectChange = (field: string, values: string[]) => {
    setFilters(prev => ({ ...prev, [field]: values }));
  };

  return (
    <div className="property-filters">
      <lyd-select 
        label="Property Type"
        value={filters.propertyType}
        onChange={(e) => handleFilterChange('propertyType', e.target.value)}
      >
        <option value="">All Types</option>
        <option value="apartment">Apartment</option>
        <option value="house">House</option>
        <option value="villa">Villa</option>
      </lyd-select>

      <lyd-select 
        label="Features" 
        multiple
        onChange={(e) => handleMultiSelectChange('features', Array.from(e.target.selectedOptions).map(o => o.value))}
      >
        <option value="balcony">Balcony</option>
        <option value="garden">Garden</option>
        <option value="parking">Parking</option>
      </lyd-select>
    </div>
  );
}

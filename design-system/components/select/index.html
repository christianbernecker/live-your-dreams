Next.js Usage

// app/properties/search/page.tsx
'use client';

import { useEffect, useState } from 'react';

export default function PropertySearchPage() {
  const [filters, setFilters] = useState({});
  
  useEffect(() => {
    import('@liveyourdreams/design-system');
  }, []);

  const handleFilterChange = (e) => {
    setFilters(prev => ({
      ...prev,
      [e.target.getAttribute('name')]: e.detail.value
    }));
  };

  return (
    <div className="space-y-6">
      <lyd-select 
        name="propertyType"
        label="Property Type"
        placeholder="Choose property type..."
        onLydSelectChange={handleFilterChange}
      >
        <lyd-option value="apartment">Apartment</lyd-option>
        <lyd-option value="house">House</lyd-option>
        <lyd-option value="penthouse">Penthouse</lyd-option>
      </lyd-select>
      
      <lyd-select 
        name="location"
        label="Location"
        placeholder="Search locations..."
        searchable
        onLydSelectChange={handleFilterChange}
      >
        <lyd-option value="schwabing">Munich - Schwabing</lyd-option>
        <lyd-option value="maxvorstadt">Munich - Maxvorstadt</lyd-option>
      </lyd-select>
      
      <lyd-select 
        name="features"
        label="Features"
        placeholder="Select features..."
        multiple
        onLydSelectChange={handleFilterChange}
      >
        <lyd-option value="balcony">Balcony</lyd-option>
        <lyd-option value="garage">Garage</lyd-option>
        <lyd-option value="elevator">Elevator</lyd-option>
      </lyd-select>
    </div>
  );
}

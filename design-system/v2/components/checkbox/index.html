Next.js Integration

// components/CheckboxGroup.tsx
import { useState, useEffect } from 'react';

interface CheckboxOption {
  value: string;
  label: string;
  description?: string;
  disabled?: boolean;
}

interface CheckboxGroupProps {
  name: string;
  options: CheckboxOption[];
  values?: string[];
  defaultValues?: string[];
  size?: 'small' | 'medium' | 'large';
  error?: boolean;
  indeterminate?: boolean;
  onChange?: (values: string[]) => void;
  className?: string;
}

export const CheckboxGroup = ({
  name,
  options,
  values,
  defaultValues = [],
  size = 'medium',
  error = false,
  indeterminate = false,
  onChange,
  className = ''
}: CheckboxGroupProps) => {
  const [selectedValues, setSelectedValues] = useState(values || defaultValues);
  
  useEffect(() => {
    if (values !== undefined) {
      setSelectedValues(values);
    }
  }, [values]);

  const handleChange = (optionValue: string, isChecked: boolean) => {
    let newValues;
    if (isChecked) {
      newValues = [...selectedValues, optionValue];
    } else {
      newValues = selectedValues.filter(v => v !== optionValue);
    }
    
    if (values === undefined) {
      setSelectedValues(newValues);
    }
    onChange?.(newValues);
  };

  const groupClasses = \`
    lyd-checkbox-group 
    \${size !== 'medium' ? \`lyd-checkbox-\${size}\` : ''} 
    \${error ? 'lyd-checkbox-error' : ''}
    \${className}
  \`.trim();

  return (
    <div className={groupClasses} data-group={name}>
      {options.map((option) => {
        const isChecked = selectedValues.includes(option.value);
        return (
          <div key={option.value} className="lyd-checkbox-item">
            <div 
              className={\`lyd-checkbox-input \${isChecked ? 'checked' : ''} \${indeterminate ? 'indeterminate' : ''} \${option.disabled ? 'disabled' : ''}\`}
              data-value={option.value}
              onClick={() => !option.disabled && handleChange(option.value, !isChecked)}
            />
            <label 
              className="lyd-checkbox-label" 
              onClick={() => !option.disabled && handleChange(option.value, !isChecked)}
            >
              <div className="lyd-checkbox-label-text">{option.label}</div>
              {option.description && (
                <div className="lyd-checkbox-label-description">{option.description}</div>
              )}
            </label>
          </div>
        );
      })}
    </div>
  );
};

// Usage Example:
export default function PropertyForm() {
  const [selectedFeatures, setSelectedFeatures] = useState(['balcony', 'garage']);
  
  const featureOptions = [
    { value: 'balcony', label: 'Balcony', description: 'Private outdoor space' },
    { value: 'garage', label: 'Garage', description: 'Covered parking space' },
    { value: 'pool', label: 'Swimming Pool', description: 'Private or shared pool access' },
    { value: 'elevator', label: 'Elevator', description: 'Building elevator access' }
  ];

  return (
    <form>
      <CheckboxGroup
        name="property-features"
        options={featureOptions}
        values={selectedFeatures}
        onChange={setSelectedFeatures}
      />
    </form>
  );
}

Accessibility & Best Practices

Keyboard Navigation

Navigate through checkbox options using Tab and Space keys for easy keyboard access and selection.

Screen Reader Support

Checkboxes include proper ARIA labels and group associations for comprehensive screen reader compatibility.

Focus Indicators

Clear visual focus indicators help users understand which checkbox is currently focused for interaction.

High Contrast

Checkboxes maintain visibility and usability in high contrast modes and for users with visual impairments.

Indeterminate State

Support for indeterminate state when dealing with nested selections and partial selections.

Multi-Select Clarity

Clear visual distinction from radio buttons to indicate multiple selection capability.

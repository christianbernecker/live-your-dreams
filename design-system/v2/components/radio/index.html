Radio

Radio buttons allow users to select one option from a set of mutually exclusive options. They work in groups where only one option can be selected at a time.

Next.js Integration

// components/RadioGroup.tsx
import { ReactNode, useState, useEffect } from 'react';

interface RadioOption {
  value: string;
  label: string;
  description?: string;
  disabled?: boolean;
}

interface RadioGroupProps {
  name: string;
  options: RadioOption[];
  value?: string;
  defaultValue?: string;
  size?: 'small' | 'medium' | 'large';
  error?: boolean;
  onChange?: (value: string) => void;
  className?: string;
}

export const RadioGroup = ({
  name,
  options,
  value,
  defaultValue,
  size = 'medium',
  error = false,
  onChange,
  className = ''
}: RadioGroupProps) => {
  const [selectedValue, setSelectedValue] = useState(value || defaultValue || '');
  
  useEffect(() => {
    if (value !== undefined) {
      setSelectedValue(value);
    }
  }, [value]);

  const handleChange = (optionValue: string) => {
    if (value === undefined) {
      setSelectedValue(optionValue);
    }
    onChange?.(optionValue);
  };

  const groupClasses = \`
    lyd-radio-group 
    \${size !== 'medium' ? \`lyd-radio-\${size}\` : ''} 
    \${error ? 'lyd-radio-error' : ''}
    \${className}
  \`.trim();

  return (
    <div className={groupClasses} data-group={name}>
      {options.map((option) => (
        <div key={option.value} className="lyd-radio-item">
          <div 
            className={\`lyd-radio-input \${selectedValue === option.value ? 'checked' : ''} \${option.disabled ? 'disabled' : ''}\`}
            data-value={option.value}
            onClick={() => !option.disabled && handleChange(option.value)}
          />
          <label className="lyd-radio-label" onClick={() => !option.disabled && handleChange(option.value)}>
            <div className="lyd-radio-label-text">{option.label}</div>
            {option.description && (
              <div className="lyd-radio-label-description">{option.description}</div>
            )}
          </label>
        </div>
      ))}
    </div>
  );
};

// Usage Example:
export default function PropertyForm() {
  const [propertyType, setPropertyType] = useState('apartment');
  
  const propertyOptions = [
    { value: 'apartment', label: 'Apartment', description: 'Modern city apartment with amenities' },
    { value: 'house', label: 'House', description: 'Single-family house with garden' },
    { value: 'penthouse', label: 'Penthouse', description: 'Luxury penthouse with panoramic views' },
    { value: 'commercial', label: 'Commercial', description: 'Office or retail space' }
  ];

  return (
    <form>
      <RadioGroup
        name="property-type"
        options={propertyOptions}
        value={propertyType}
        onChange={setPropertyType}
      />
    </form>
  );
}

Accessibility & Best Practices

Keyboard Navigation

Navigate through radio options using Tab, Space, and arrow keys for easy keyboard access.

Screen Reader Support

Radio buttons include proper ARIA labels and group associations for screen reader compatibility.

Focus Indicators

Clear visual focus indicators help users understand which radio button is currently selected.

High Contrast

Radio buttons maintain visibility and usability in high contrast modes and for users with visual impairments.

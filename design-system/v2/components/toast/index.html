JavaScript Integration

// Toast Management System
class LYDToast {
    constructor(container = '#toastContainer') {
        this.container = document.querySelector(container);
        this.toasts = new Map();
        this.autoId = 0;
    }

    show(type, title, message, options = {}) {
        const id = options.id || `toast-${++this.autoId}`;
        const duration = options.duration || 5000;
        const closable = options.closable !== false;

        const toast = this.createToast(type, title, message, id, closable);
        this.container.appendChild(toast);
        this.toasts.set(id, toast);

        // Auto-dismiss
        if (duration > 0) {
            setTimeout(() => this.hide(id), duration);
        }

        return id;
    }

    createToast(type, title, message, id, closable) {
        const toast = document.createElement('div');
        toast.className = `lyd-toast ${type}`;
        toast.dataset.id = id;

        const icons = {
            success: 'M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z',
            error: 'M12 8v4m0 4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z',
            warning: 'M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-2.5L13.732 4c-.77-.833-1.964-.833-2.732 0L3.732 16.5c-.77.833.192 2.5 1.732 2.5z',
            info: 'M13 16h-1v-4h-1m1-4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z'
        };

        toast.innerHTML = `
            <svg class="toast-icon ${type}" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="${icons[type]}"/>
            </svg>
            <div class="toast-content">
                <div class="toast-title">${title}</div>
                <div class="toast-message">${message}</div>
            </div>
            ${closable ? '<button class="toast-close">×</button>' : ''}
        `;

        if (closable) {
            toast.querySelector('.toast-close').onclick = () => this.hide(id);
        }

        return toast;
    }

    hide(id) {
        const toast = this.toasts.get(id);
        if (toast) {
            toast.classList.add('removing');
            setTimeout(() => {
                if (toast.parentNode) {
                    toast.parentNode.removeChild(toast);
                }
                this.toasts.delete(id);
            }, 300);
        }
    }

    // Real Estate specific methods
    propertyListed(propertyTitle) {
        return this.show('success', 'Property Listed Successfully', 
            `${propertyTitle} has been published and is now visible to potential buyers.`);
    }

    viewingScheduled(propertyTitle, date) {
        return this.show('info', 'Viewing Scheduled', 
            `Property viewing for ${propertyTitle} scheduled for ${date}.`);
    }

    offerSubmitted(amount, propertyTitle) {
        return this.show('success', 'Offer Submitted', 
            `Your offer of €${amount.toLocaleString()} for ${propertyTitle} has been submitted.`);
    }

    priceDropAlert(propertyTitle, oldPrice, newPrice) {
        const savings = oldPrice - newPrice;
        return this.show('info', 'Price Drop Alert', 
            `${propertyTitle} price reduced by €${savings.toLocaleString()}. New price: €${newPrice.toLocaleString()}.`);
    }
}

React/Next.js Integration

// components/Toast.tsx
import React, { createContext, useContext, useState, useCallback } from 'react';

interface Toast {
  id: string;
  type: 'success' | 'error' | 'warning' | 'info';
  title: string;
  message: string;
  duration?: number;
}

interface ToastContextType {
  toasts: Toast[];
  showToast: (type: Toast['type'], title: string, message: string, duration?: number) => void;
  hideToast: (id: string) => void;
  // Real Estate specific methods
  propertyListed: (propertyTitle: string) => void;
  viewingScheduled: (propertyTitle: string, date: string) => void;
  offerSubmitted: (amount: number, propertyTitle: string) => void;
}

const ToastContext = createContext<ToastContextType | undefined>(undefined);

export const ToastProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [toasts, setToasts] = useState<Toast[]>([]);

  const showToast = useCallback((type: Toast['type'], title: string, message: string, duration = 5000) => {
    const id = `toast-${Date.now()}-${Math.random()}`;
    const newToast: Toast = { id, type, title, message, duration };
    
    setToasts(prev => [...prev, newToast]);
    
    if (duration > 0) {
      setTimeout(() => hideToast(id), duration);
    }
  }, []);

  const hideToast = useCallback((id: string) => {
    setToasts(prev => prev.filter(toast => toast.id !== id));
  }, []);

  // Real Estate specific methods
  const propertyListed = useCallback((propertyTitle: string) => {
    showToast('success', 'Property Listed Successfully', 
      `${propertyTitle} has been published and is now visible to potential buyers.`);
  }, [showToast]);

  const viewingScheduled = useCallback((propertyTitle: string, date: string) => {
    showToast('info', 'Viewing Scheduled', 
      `Property viewing for ${propertyTitle} scheduled for ${date}.`);
  }, [showToast]);

  const offerSubmitted = useCallback((amount: number, propertyTitle: string) => {
    showToast('success', 'Offer Submitted', 
      `Your offer of €${amount.toLocaleString()} for ${propertyTitle} has been submitted.`);
  }, [showToast]);

  return (
    <ToastContext.Provider value={{ 
      toasts, 
      showToast, 
      hideToast, 
      propertyListed, 
      viewingScheduled, 
      offerSubmitted 
    }}>
      {children}
      <ToastContainer />
    </ToastContext.Provider>
  );
};

export const useToast = () => {
  const context = useContext(ToastContext);
  if (!context) {
    throw new Error('useToast must be used within a ToastProvider');
  }
  return context;
};

// Usage in components
const PropertyForm = () => {
  const { propertyListed, showToast } = useToast();
  
  const handleSubmit = async (formData) => {
    try {
      await submitProperty(formData);

    }
  };
};

Property Management Workflows

// Property Management Toast Integration
class PropertyToastManager {
    constructor(toastSystem) {
        this.toast = toastSystem;
    }

    // Listing Management
    onPropertyListed(property) {
        this.toast.show('success', 'Property Listed Successfully', 
            `${property.title} in ${property.location} is now live with ${property.images.length} photos.`);
    }

    onPropertyUpdated(property, changes) {
        const changeText = Object.keys(changes).join(', ');
        this.toast.show('info', 'Property Updated', 
            `${property.title} has been updated. Changes: ${changeText}.`);
    }

    // Financial Notifications
    onOfferReceived(offer) {
        this.toast.show('success', 'New Offer Received', 
            `€${offer.amount.toLocaleString()} offer received for ${offer.property.title}.`, 
            { duration: 8000 });
    }

    onPriceReduction(property, oldPrice, newPrice) {
        const reduction = oldPrice - newPrice;
        const percentage = Math.round((reduction / oldPrice) * 100);
        this.toast.show('warning', 'Price Reduced', 
            `${property.title} price reduced by ${percentage}% (€${reduction.toLocaleString()}).`);
    }

    // Market Intelligence
    onMarketUpdate(area, trend) {
        this.toast.show('info', 'Market Update', 
            `${area} market ${trend.direction}: Average price ${trend.change}.`);
    }

    // System & Integration
    onCRMSync(recordsUpdated) {
        this.toast.show('success', 'CRM Sync Complete', 
            `${recordsUpdated} property records synchronized successfully.`);
    }

    onMaintenanceAlert(startTime, duration) {
        this.toast.show('warning', 'Scheduled Maintenance', 
            `System maintenance from ${startTime} for ${duration}. Limited functionality expected.`, 
            { duration: 10000 });
    }
}

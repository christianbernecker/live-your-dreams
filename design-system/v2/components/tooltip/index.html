React/Next.js Component

// components/Tooltip.tsx
import { ReactNode, useState } from 'react';

interface TooltipProps {
  children: ReactNode;
  content: string | ReactNode;
  position?: 'top' | 'bottom' | 'left' | 'right';
  variant?: 'default' | 'primary' | 'success' | 'warning' | 'error';
  multiline?: boolean;
  className?: string;
}

export const Tooltip = ({
  children,
  content,
  position = 'top',
  variant = 'default',
  multiline = false,
  className = ''
}: TooltipProps) => {
  const [isVisible, setIsVisible] = useState(false);
  
  const variantClass = variant !== 'default' ? variant : '';
  const multilineClass = multiline ? 'multiline' : '';
  
  return (
    <div 
      className={`lyd-tooltip ${className}`}
      onMouseEnter={() => setIsVisible(true)}
      onMouseLeave={() => setIsVisible(false)}
    >
      {children}
      {isVisible && (
        <div className={`lyd-tooltip-content ${position} ${variantClass} ${multilineClass}`}>
          {content}
        </div>
      )}
    </div>
  );
};

// Usage Examples
<Tooltip content="Total living area">
  <span>250 m²</span>
</Tooltip>

<Tooltip 
  content="This property features luxury amenities including a private pool"
  position="bottom"
  multiline
>
  <button>View Details</button>
</Tooltip>

<Tooltip content="Available" variant="success" position="right">
  <StatusIndicator />
</Tooltip>

.lyd-tooltip-content Tooltip content box
.top, .bottom, .left, .right Position modifiers
.multiline Multi-line text support
.primary Primary color variant
.success Success color variant
.warning Warning color variant
.error Error color variant
.show Force tooltip visible

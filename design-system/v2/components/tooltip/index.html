Next.js Integration

import { useState } from 'react';

interface TooltipProps {
  label: string;
  content: string;
  className?: string;
}

export const LydTooltip: React.FC<TooltipProps> = ({ 
  label, 
  content, 
  className = '' 
}) => {
  const [isVisible, setIsVisible] = useState(false);

  return (
    <div 
      className={`lyd-tooltip-container ${className}`}
      onMouseEnter={() => setIsVisible(true)}
      onMouseLeave={() => setIsVisible(false)}
    >
      <span>{label}
        <span className="lyd-tooltip-trigger">
          <svg width="16" height="16" viewBox="0 0 24 24" fill="currentColor">
            <circle cx="12" cy="12" r="10" fill="var(--lyd-royal-blue)" opacity="0.1"/>
            <circle cx="12" cy="12" r="10" fill="none" stroke="currentColor" strokeWidth="1.5"/>
            <path d="M12 7.5h.01" stroke="currentColor" strokeWidth="2" strokeLinecap="round"/>
            <path d="M11 11h1v5h1" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round"/>
          </svg>
        </span>
      </span>
      <div className={`lyd-tooltip ${isVisible ? 'visible' : ''}`}>
        {content}
      </div>
    </div>
  );
};

// Usage Example
export default function PropertyCard() {
  return (
    <div className="property-details">
      <LydTooltip 
        label="Energy Efficiency Rating"
        content="Energy efficiency rating from A+ (most efficient) to H (least efficient). This property has an A+ rating."
      />
      
      <LydTooltip 
        label="Monthly Maintenance Costs"
        content="Includes property management, building maintenance, insurance, and shared utilities."
      />
    </div>
  );
}

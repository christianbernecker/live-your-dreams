React/Next.js Integration

// components/Select.tsx
import { useState, useRef, useEffect } from 'react';

interface SelectOption {
  value: string;
  label: string;
  icon?: React.ReactNode;
  meta?: string;
  group?: string;
}

interface SelectProps {
  options: SelectOption[];
  value?: string | string[];
  placeholder?: string;
  searchable?: boolean;
  multiple?: boolean;
  disabled?: boolean;
  error?: boolean;
  success?: boolean;
  loading?: boolean;
  onChange: (value: string | string[]) => void;
  onSearch?: (query: string) => void;
}

export const Select = ({
  options,
  value,
  placeholder = "Select option...",
  searchable = false,
  multiple = false,
  disabled = false,
  error = false,
  success = false,
  loading = false,
  onChange,
  onSearch
}: SelectProps) => {
  const [isOpen, setIsOpen] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedValues, setSelectedValues] = useState<string[]>(
    multiple ? (value as string[] || []) : (value ? [value as string] : [])
  );

  const selectRef = useRef<HTMLDivElement>(null);

  const filteredOptions = options.filter(option =>
    searchable ? option.label.toLowerCase().includes(searchQuery.toLowerCase()) : true
  );

  const handleSelect = (optionValue: string) => {
    if (multiple) {
      const newValues = selectedValues.includes(optionValue)
        ? selectedValues.filter(v => v !== optionValue)
        : [...selectedValues, optionValue];
      setSelectedValues(newValues);
      onChange(newValues);
    } else {
      setSelectedValues([optionValue]);
      onChange(optionValue);
      setIsOpen(false);
    }
  };

  const removeTag = (valueToRemove: string) => {
    const newValues = selectedValues.filter(v => v !== valueToRemove);
    setSelectedValues(newValues);
    onChange(newValues);
  };

      return (
    <div className="lyd-select" ref={selectRef}>
      <div 
        className={`lyd-select-trigger ${isOpen ? 'active' : ''} ${multiple ? 'lyd-multi-select' : ''}`}
        onClick={() => !disabled && setIsOpen(!isOpen)}
        tabIndex={0}
      >
        {multiple && selectedValues.map(val => {
          const option = options.find(opt => opt.value === val);
          return (
            <div key={val} className="lyd-select-tag">
              {option?.label}
              <div 
                className="lyd-select-tag-remove"
                onClick={(e) => {
                  e.stopPropagation();
                  removeTag(val);
                }}
              >
                ×
              </div>
            </div>
          );
        })}
        
        {searchable ? (
          <input
            type="text"
            className="lyd-select-search"
            placeholder={placeholder}
            value={searchQuery}
            onChange={(e) => {
              setSearchQuery(e.target.value);
              onSearch?.(e.target.value);
            }}
          />
        ) : (
          <span className="lyd-select-value">
            {selectedValues.length === 0 ? placeholder : 
             multiple ? `${selectedValues.length} selected` :
             options.find(opt => opt.value === selectedValues[0])?.label}
          </span>
        )}
        
        <svg className="lyd-select-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M19 9l-7 7-7-7"/>
        </svg>
      </div>
      
      {isOpen && (
        <div className="lyd-select-dropdown active">
          {loading ? (
            <div className="lyd-select-loading">
              <div className="lyd-select-spinner"></div>
              Loading options...
            </div>
          ) : (
            filteredOptions.map((option, index) => (
              <div
                key={option.value}
                className={`lyd-select-option ${selectedValues.includes(option.value) ? 'selected' : ''}`}
                onClick={() => handleSelect(option.value)}
              >
                {option.icon && <span className="lyd-property-icon">{option.icon}</span>}
                <div className="lyd-property-details">
                  <div className="lyd-property-name">{option.label}</div>
                  {option.meta && <div className="lyd-property-meta">{option.meta}</div>}
                </div>
              </div>
            ))
          )}
        </div>
      )}
    </div>
  );
};

Advanced Features & Patterns

Async Data Loading Pattern

// Async Select with Debounced Search
const AsyncLocationSelect = () => {
  const [options, setOptions] = useState([]);
  const [loading, setLoading] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  
  // Debounced search
  const debouncedSearch = useMemo(
    () => debounce(async (query: string) => {
      setLoading(true);
      try {
        const response = await fetch(`/api/locations?q=${query}`);
        const data = await response.json();
        setOptions(data.map(item => ({
          value: item.id,
          label: item.name,
          metadata: { 
            region: item.region,
            postalCode: item.postalCode,
            propertyCount: item.propertyCount
          }
        })));
      } finally {
        setLoading(false);
      }
    }, 300),
    []
  );
  
  useEffect(() => {
    if (searchQuery) {
      debouncedSearch(searchQuery);
    }
  }, [searchQuery]);
  
  return (
    <Select
      options={options}
      loading={loading}
      searchable
      asyncSearch
      onSearch={setSearchQuery}
      placeholder="Type to search locations..."
      renderOption={(option) => (
        <div className="location-option">
          <div className="location-name">{option.label}</div>
          <div className="location-meta">
            {option.metadata.region} • {option.metadata.postalCode}
            <span className="property-count">
              {option.metadata.propertyCount} properties
            </span>
          </div>
        </div>
      )}
    />
  );
};

Virtualization for Large Datasets

// Virtual Scrolling for 1000+ Options
import { FixedSizeList } from 'react-window';

const VirtualizedSelect = ({ options }) => {
  const ITEM_HEIGHT = 40;
  const MAX_HEIGHT = 300;
  
  const Row = ({ index, style }) => {
    const option = options[index];
    return (
      <div style={style} className="lyd-select-option">
        {option.label}
      </div>
    );
  };
  
  return (
    <Select
      options={options}
      renderDropdown={() => (
        <FixedSizeList
          height={Math.min(options.length * ITEM_HEIGHT, MAX_HEIGHT)}
          itemCount={options.length}
          itemSize={ITEM_HEIGHT}
          width="100%"
        >
          {Row}
        </FixedSizeList>
      )}
    />
  );
};

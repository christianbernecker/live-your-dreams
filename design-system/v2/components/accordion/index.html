React/Next.js Integration

// components/Accordion.tsx
import { ReactNode, useState } from 'react';

interface AccordionItemProps {
  title: string;
  children: ReactNode;
  defaultOpen?: boolean;
}

interface AccordionProps {
  variant?: 'default' | 'compact' | 'bordered' | 'flush' | 'glassmorphism';
  className?: string;
  children: ReactNode;
}

export const AccordionItem = ({ 
  title, 
  children, 
  defaultOpen = false 
}: AccordionItemProps) => {
  const [isOpen, setIsOpen] = useState(defaultOpen);

  return (
    <div className="lyd-accordion-item">
      <button 
        className={`lyd-accordion-header ${isOpen ? 'active' : ''}`}
        onClick={() => setIsOpen(!isOpen)}
        aria-expanded={isOpen}
      >
        <span>{title}</span>
        <svg className="lyd-accordion-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M19 9l-7 7-7-7"/>
        </svg>
      </button>
      <div className={`lyd-accordion-content ${isOpen ? 'active' : ''}`}>
        <div className="lyd-accordion-body">
          {children}
        </div>
      </div>
    </div>
  );
};

export const Accordion = ({ 
  variant = 'default', 
  className = '',
  children 
}: AccordionProps) => {
  const accordionClasses = `
    lyd-accordion 
    ${variant !== 'default' ? variant : ''} 
    ${className}
  `.trim();

  return (
    <div className={accordionClasses}>
      {children}
    </div>
  );
};

// Usage Example
export default function PropertyDetails() {
  return (
    <Accordion variant="bordered">
      <AccordionItem title="Property Features">
        <p>Detailed property specifications and amenities.</p>
      </AccordionItem>
      
      <AccordionItem title="Location Details" defaultOpen>
        <p>Neighborhood information and nearby facilities.</p>
      </AccordionItem>
      
      <AccordionItem title="Investment Potential">
        <p>Market analysis and rental yield information.</p>
      </AccordionItem>
    </Accordion>
  );
}

Accessibility & Best Practices

Keyboard Navigation

  • Tab to navigate between accordion headers
  • Enter/Space to toggle accordion sections
  • Arrow keys for vertical navigation
  • Focus indicators on all interactive elements
  • Logical tab order throughout component

Screen Reader Support

  • Proper ARIA attributes (expanded, controls)
  • Semantic HTML structure with buttons
  • Clear heading hierarchy
  • Descriptive button labels
  • State announcements for expand/collapse

Visual Accessibility

  • WCAG AA contrast ratios (4.5:1 minimum)
  • Clear focus indicators
  • Sufficient color differentiation
  • Readable typography scales
  • Animation respects reduced motion preferences

Interactive States

  • Hover feedback for all clickable elements
  • Active state during interaction
  • Smooth expand/collapse animations
  • Icon rotation indicates state change
  • Loading states for dynamic content

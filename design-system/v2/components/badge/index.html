// components/Badge.tsx
import { ReactNode } from 'react';

interface BadgeProps {
  variant?: 'primary' | 'success' | 'error' | 'warning' | 'secondary' | 'accent';
  style?: 'filled' | 'outlined' | 'soft';
  size?: 'small' | 'medium' | 'large';
  icon?: ReactNode;
  dismissible?: boolean;
  clickable?: boolean;
  children: ReactNode;
  onClick?: () => void;
  onDismiss?: () => void;
  className?: string;
}

export const Badge = ({
  variant = 'primary',
  style = 'filled',
  size = 'medium',
  icon,
  dismissible = false,
  clickable = false,
  children,
  onClick,
  onDismiss,
  className = ''
}: BadgeProps) => {
  const badgeClasses = \`
    lyd-badge 
    \${variant}
    \${style !== 'filled' ? style : ''}
    \${size !== 'medium' ? size : ''}
    \${clickable ? 'clickable' : ''}
    \${className}
  \`.trim();

  const handleClick = () => {
    if (clickable && onClick) {
      onClick();
    }
  };

  const handleDismiss = (e: React.MouseEvent) => {
    e.stopPropagation();
    onDismiss?.();
  };

  return (
    <span 
      className={badgeClasses}
      onClick={handleClick}
      role={clickable ? 'button' : undefined}
      tabIndex={clickable ? 0 : undefined}
      onKeyDown={(e) => {
        if (clickable && (e.key === ' ' || e.key === 'Enter')) {
          e.preventDefault();
          handleClick();
        }
      }}
    >
      {icon && <span className="lyd-badge-icon">{icon}</span>}
      {children}
      {dismissible && (
        <button 
          className="lyd-badge-close" 
          onClick={handleDismiss}
          aria-label="Remove badge"
        >
          Ã
        </button>
      )}
    </span>
  );
};

// Property Status Badge Component
interface PropertyStatusBadgeProps {
  status: 'available' | 'reserved' | 'sold' | 'under-review';
  className?: string;
}

export const PropertyStatusBadge = ({ 
  status, 
  className = '' 
}: PropertyStatusBadgeProps) => {
  const statusConfig = {
    available: { variant: 'success' as const, icon: 'â', text: 'Available' },
    reserved: { variant: 'warning' as const, icon: 'â³', text: 'Reserved' },
    sold: { variant: 'error' as const, icon: 'â', text: 'Sold' },
    'under-review': { variant: 'secondary' as const, icon: 'ð', text: 'Under Review' }
  };

  const config = statusConfig[status];

  return (
    <Badge
      variant={config.variant}
      icon={config.icon}
      className={className}
    >
      {config.text}
    </Badge>
  );
};

Accessibility & Best Practices

Semantic Markup

Proper role attributes for clickable badges and clear labeling for screen readers to understand badge purpose.

Keyboard Navigation

Full keyboard support for interactive badges with Tab navigation and Space/Enter activation.

Color Independence

Icons and text content provide meaning beyond color, ensuring accessibility for all users regardless of color vision.

Focus Indicators

Clear visual focus indicators for interactive badges, ensuring keyboard navigation is visible and intuitive.

Dismissible Actions

Proper labeling for close buttons and clear indication of dismissible functionality for screen readers.

Content Clarity

Concise, descriptive text that clearly communicates the badge's purpose and status to all users.

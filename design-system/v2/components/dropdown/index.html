React/Next.js Integration

// components/Dropdown.tsx
import { useEffect, useRef, useState } from 'react';

interface DropdownProps {
  label: string;
  items: { label: string; onSelect: () => void }[];
}

export const Dropdown = ({ label, items }: DropdownProps) => {
  const [open, setOpen] = useState(false);
  const ref = useRef(null);

  useEffect(() => {
    const onDoc = (e: MouseEvent) => { if (ref.current && !ref.current.contains(e.target as Node)) setOpen(false); };
    document.addEventListener('click', onDoc);
    return () => document.removeEventListener('click', onDoc);
  }, []);

  return (
    <div className="lyd-dropdown" ref={ref}>
      <button className="lyd-button" aria-haspopup="menu" aria-expanded={open} onClick={() => setOpen(v => !v)}>
        {label}
      </button>
      {open && (
        <div className="lyd-dropdown-menu" role="menu">
          {items.map((it, i) => (
            <button role="menuitem" className="lyd-dropdown-item" key={i} onClick={() => { setOpen(false); it.onSelect(); }}>
              {it.label}
            </button>
          ))}
        </div>
      )}
    </div>
  );
};

Accessibility & Best Practices

Keyboard Navigation

Full keyboard support with Enter/Space to open, arrow keys to navigate, and Escape to close dropdown.

Screen Reader Support

ARIA attributes like aria-haspopup and aria-expanded provide clear context for assistive technologies.

Focus Management

Proper focus handling ensures keyboard users can navigate efficiently through dropdown options.

Visual Clarity

High contrast ratios, clear hover states, and consistent spacing for optimal usability.

// components/Switch.tsx
import { useState } from 'react';

interface SwitchProps {
  id?: string;
  checked?: boolean;
  defaultChecked?: boolean;
  size?: 'small' | 'medium' | 'large';
  color?: 'primary' | 'success' | 'warning' | 'error';
  disabled?: boolean;
  label?: string;
  description?: string;
  onChange?: (checked: boolean) => void;
  className?: string;
}

export const Switch = ({
  id,
  checked,
  defaultChecked = false,
  size = 'medium',
  color = 'primary',
  disabled = false,
  label,
  description,
  onChange,
  className = ''
}: SwitchProps) => {
  const [isChecked, setIsChecked] = useState(checked ?? defaultChecked);

  const handleToggle = () => {
    if (disabled) return;
    
    const newChecked = !isChecked;
    if (checked === undefined) {
      setIsChecked(newChecked);
    }
    onChange?.(newChecked);
  };

  const switchClasses = \`
    lyd-switch 
    \${size !== 'medium' ? size : ''}
    \${color !== 'primary' ? color : ''}
    \${(checked ?? isChecked) ? 'checked' : ''}
    \${disabled ? 'disabled' : ''}
    \${className}
  \`.trim();

  return (
    <div className="lyd-switch-item">
      <div 
        className={switchClasses}
        onClick={handleToggle}
        role="switch"
        aria-checked={checked ?? isChecked}
        aria-disabled={disabled}
        tabIndex={disabled ? -1 : 0}
        onKeyDown={(e) => {
          if (e.key === ' ' || e.key === 'Enter') {
            e.preventDefault();
            handleToggle();
          }
        }}
      />
      {(label || description) && (
        <label className="lyd-switch-label" onClick={handleToggle}>
          {label && (
            <div className="lyd-switch-label-text">{label}</div>
          )}
          {description && (
            <div className="lyd-switch-label-description">{description}</div>
          )}
        </label>
      )}
    </div>
  );
};

// Usage Example:
export default function PropertySettings() {
  const [settings, setSettings] = useState({
    virtualTour: true,
    furnished: false,
    parkingIncluded: true,
    petsAllowed: false
  });

  const updateSetting = (key: keyof typeof settings, value: boolean) => {
    setSettings(prev => ({ ...prev, [key]: value }));
  };

  return (
    <div className="lyd-switch-group">
      <Switch
        label="Virtual Tour Available"
        description="360° virtual property walkthrough"
        checked={settings.virtualTour}
        onChange={(checked) => updateSetting('virtualTour', checked)}
      />
      
      <Switch
        label="Furnished Property"
        description="Property comes fully furnished"
        checked={settings.furnished}
        onChange={(checked) => updateSetting('furnished', checked)}
        color="success"
      />
      
      <Switch
        label="Parking Included"
        description="Dedicated parking space available"
        checked={settings.parkingIncluded}
        onChange={(checked) => updateSetting('parkingIncluded', checked)}
        size="large"
      />
    </div>
  );
}

Class Description
.lyd-switch Base switch class with premium styling
.lyd-switch-wrapper switch container with consistent spacing
.small Small switch size (14px - Typography Scale)
.large Large switch size (18px - Typography Scale)
.primary Primary switch variant with CI gradient
.success Success state with green styling
.error Error state with red styling and validation
.disabled Disabled state with reduced opacity

Accessibility & Best Practices

Keyboard Navigation

Full keyboard support with Tab navigation and Space/Enter keys for toggling switch states.

Screen Reader Support

Proper ARIA roles with switch role and aria-checked attributes for comprehensive accessibility.

Visual Feedback

Clear visual distinction between on/off states with smooth transitions and focus indicators.

Touch Friendly

Adequate touch target size (minimum 44px) for mobile devices and touch screen interactions.

Color Independence

Position-based state indication ensures functionality without relying solely on color differentiation.

Disabled States

Clear visual and programmatic disabled states with appropriate cursor and interaction handling.

// components/Slider.tsx
import { useState, useEffect, useRef } from 'react';

interface SliderProps {
  min?: number;
  max?: number;
  step?: number;
  value?: number;
  defaultValue?: number;
  // Range slider props
  minValue?: number;
  maxValue?: number;
  defaultMinValue?: number;
  defaultMaxValue?: number;
  // Appearance
  size?: 'small' | 'medium' | 'large';
  color?: 'primary' | 'success' | 'warning' | 'error';
  disabled?: boolean;
  label?: string;
  helperText?: string;
  formatValue?: (value: number) => string;
  showMarks?: boolean;
  marks?: { value: number; label: string }[];
  // Events
  onChange?: (value: number | [number, number]) => void;
  onChangeEnd?: (value: number | [number, number]) => void;
  className?: string;
}

export const Slider = ({
  min = 0,
  max = 100,
  step = 1,
  value,
  defaultValue,
  minValue,
  maxValue,
  defaultMinValue,
  defaultMaxValue,
  size = 'medium',
  color = 'primary',
  disabled = false,
  label,
  helperText,
  formatValue = (val) => val.toString(),
  showMarks = false,
  marks,
  onChange,
  onChangeEnd,
  className = ''
}: SliderProps) => {
  const isRange = minValue !== undefined || maxValue !== undefined;
  const [currentValue, setCurrentValue] = useState(() => {
    if (isRange) {
      return [
        minValue ?? defaultMinValue ?? min,
        maxValue ?? defaultMaxValue ?? max
      ] as [number, number];
    }
    return value ?? defaultValue ?? min;
  });
  
  const sliderRef = useRef(null);
  const [isDragging, setIsDragging] = useState(false);
  const [activeThumb, setActiveThumb] = useState(0);

  useEffect(() => {
    if (isRange && minValue !== undefined && maxValue !== undefined) {
      setCurrentValue([minValue, maxValue]);
    } else if (!isRange && value !== undefined) {
      setCurrentValue(value);
    }
  }, [value, minValue, maxValue, isRange]);

  const getPercentage = (val: number) => {
    return ((val - min) / (max - min)) * 100;
  };

  const getValueFromPosition = (clientX: number) => {
    if (!sliderRef.current) return min;
    
    const rect = sliderRef.current.getBoundingClientRect();
    const percentage = Math.max(0, Math.min(1, (clientX - rect.left) / rect.width));
    const rawValue = min + percentage * (max - min);
    return Math.round(rawValue / step) * step;
  };

  const handleMouseDown = (thumbIndex: number, e: React.MouseEvent) => {
    if (disabled) return;
    
    setIsDragging(true);
    setActiveThumb(thumbIndex);
    
    const handleMouseMove = (e: MouseEvent) => {
      const newValue = getValueFromPosition(e.clientX);
      
      if (isRange) {
        const [minVal, maxVal] = currentValue as [number, number];
        let newRange: [number, number];
        
        if (thumbIndex === 0) {
          newRange = [Math.min(newValue, maxVal), maxVal];
        } else {
          newRange = [minVal, Math.max(newValue, minVal)];
        }
        
        setCurrentValue(newRange);
        onChange?.(newRange);
      } else {
        setCurrentValue(newValue);
        onChange?.(newValue);
      }
    };
    
    const handleMouseUp = () => {
      setIsDragging(false);
      onChangeEnd?.(currentValue);
      document.removeEventListener('mousemove', handleMouseMove);
      document.removeEventListener('mouseup', handleMouseUp);
    };
    
    document.addEventListener('mousemove', handleMouseMove);
    document.addEventListener('mouseup', handleMouseUp);
  };

  const sliderClasses = \`
    lyd-slider lyd-slider-container
    \${size !== 'medium' ? size : ''}
    \${color !== 'primary' ? color : ''}
    \${isRange ? 'range' : ''}
    \${disabled ? 'disabled' : ''}
    \${className}
  \`.trim();

  const getDisplayValue = () => {
    if (isRange) {
      const [minVal, maxVal] = currentValue as [number, number];
      return \`\${formatValue(minVal)} - \${formatValue(maxVal)}\`;
    }
    return formatValue(currentValue as number);
  };

  const getRangeStyle = () => {
    if (isRange) {
      const [minVal, maxVal] = currentValue as [number, number];
      const minPercent = getPercentage(minVal);
      const maxPercent = getPercentage(maxVal);
      return {
        left: \`\${minPercent}%\`,
        width: \`\${maxPercent - minPercent}%\`
      };
    }
    return {
      width: \`\${getPercentage(currentValue as number)}%\`
    };
  };

  return (
    <div className="lyd-slider-group">
      {label && (
        <div className="lyd-slider-label">
          <div className="lyd-slider-label-text">{label}</div>
          <div className="lyd-slider-value">{getDisplayValue()}</div>
        </div>
      )}
      
      <div ref={sliderRef} className={sliderClasses}>
        <div className="lyd-slider-track">
          <div className="lyd-slider-range" style={getRangeStyle()} />
        </div>
        
        {isRange ? (
          <>
            <div
              className={\`lyd-slider-thumb \${isDragging && activeThumb === 0 ? 'dragging' : ''}\`}
              style={{ left: \`\${getPercentage((currentValue as [number, number])[0])}%\` }}
              onMouseDown={(e) => handleMouseDown(0, e)}
            />
            <div
              className={\`lyd-slider-thumb \${isDragging && activeThumb === 1 ? 'dragging' : ''}\`}
              style={{ left: \`\${getPercentage((currentValue as [number, number])[1])}%\` }}
              onMouseDown={(e) => handleMouseDown(1, e)}
            />
          </>
        ) : (
          <div
            className={\`lyd-slider-thumb \${isDragging ? 'dragging' : ''}\`}
            style={{ left: \`\${getPercentage(currentValue as number)}%\` }}
            onMouseDown={(e) => handleMouseDown(0, e)}
          />
        )}
      </div>
      
      {showMarks && marks && (
        <div className="lyd-slider-marks">
          {marks.map((mark, index) => (
            <span key={index}>{mark.label}</span>
          ))}
        </div>
      )}
      
      {helperText && (
        <div className="lyd-slider-helper">{helperText}</div>
      )}
    </div>
  );
};

// Usage Example:
export default function PropertyFilters() {
  const [priceRange, setPriceRange] = useState([500000, 1200000]);
  const [rooms, setRooms] = useState(3);

  const formatCurrency = (value: number) => {
    return new Intl.NumberFormat('de-DE', {
      style: 'currency',
      currency: 'EUR',
      notation: value >= 1000000 ? 'compact' : 'standard',
      maximumFractionDigits: 0
    }).format(value);
  };

  return (
    <div>
      <Slider
        label="Price Range"
        minValue={priceRange[0]}
        maxValue={priceRange[1]}
        min={200000}
        max={3000000}
        step={25000}
        formatValue={formatCurrency}
        onChange={(value) => setPriceRange(value as [number, number])}
        helperText="Filter properties within your budget range"
      />
      
      <Slider
        label="Number of Rooms"
        value={rooms}
        min={1}
        max={8}
        step={1}
        formatValue={(val) => \`\${val} room\${val !== 1 ? 's' : ''}\`}
        onChange={(value) => setRooms(value as number)}
        color="success"
        size="large"
      />
    </div>
  );
}

API Reference

Class Description
.lyd-slider Base slider class with premium styling
.lyd-slider-wrapper slider container with consistent spacing
.small Small slider size (14px - Typography Scale)
.large Large slider size (18px - Typography Scale)
.primary Primary slider variant with CI gradient
.success Success state with green styling
.error Error state with red styling and validation
.disabled Disabled state with reduced opacity

Accessibility & Best Practices

Keyboard Navigation

Full keyboard support with Tab navigation and Arrow keys for precise value adjustment without mouse interaction.

Screen Reader Support

ARIA slider role with proper min/max/value attributes and value announcements for screen reader users.

Touch Friendly

Large touch targets (minimum 44px) with drag gestures optimized for mobile and tablet devices.

Visual Feedback

Clear visual indicators for active state, hover effects, and value changes with smooth animations.

Value Formatting

Contextual value formatting (currency, percentage, units) with proper localization support.

Range Validation

Built-in min/max validation with step constraints to ensure valid value selection and prevent errors.

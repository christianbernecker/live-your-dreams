Next.js Integration

// components/Avatar.tsx
interface AvatarProps {
  src?: string;
  alt?: string;
  initials?: string;
  size?: 'xs' | 'sm' | 'md' | 'lg' | 'xl' | 'xxl';
  status?: 'online' | 'away' | 'busy' | 'offline';
  className?: string;
}

export const Avatar = ({ src, alt, initials, size = 'md', status, className = '' }: AvatarProps) => {
  const avatarClasses = \`lyd-avatar \${size} \${className}\`.trim();
  
  return (
    <div className={avatarClasses}>
      {src ? (
        <img src={src} alt={alt || 'Avatar'} />
      ) : initials ? (
        <div className="lyd-avatar-fallback">{initials}</div>
      ) : (
        <UserIcon className="lyd-avatar-icon" />
      )}
      {status && <div className={\`lyd-avatar-status \${status}\`} />}
    </div>
  );
};

React/Next.js Integration

// components/Modal.tsx
import { ReactNode, useEffect } from 'react';

type ModalSize = 'small' | 'medium' | 'large';
type ModalVariant = 'default' | 'glassmorphism';

interface ModalProps {
  isOpen: boolean;
  onClose: () => void;
  title: string;
  size?: ModalSize;
  variant?: ModalVariant;
  children: ReactNode;
}

export const Modal = ({
  isOpen,
  onClose,
  title,
  size = 'medium',
  variant = 'default',
  children
}: ModalProps) => {
  useEffect(() => {
    if (!isOpen) return;
    document.body.style.overflow = 'hidden';
    const onKey = (e: KeyboardEvent) => e.key === 'Escape' && onClose();
    document.addEventListener('keydown', onKey);
    return () => { document.body.style.overflow = ''; document.removeEventListener('keydown', onKey); };
  }, [isOpen, onClose]);

  if (!isOpen) return null;

  return (
    <div className="lyd-modal active" role="dialog" aria-modal="true">
      <div className="lyd-modal-backdrop" onClick={onClose} />
      <div className={`lyd-modal-content ${size} ${variant}`}>
        <div className="lyd-modal-header">
          <h3 className="lyd-modal-title">{title}</h3>
          <button className="lyd-modal-close" onClick={onClose} aria-label="Close">Ã</button>
        </div>
        <div className="lyd-modal-body">{children}</div>
        <div className="lyd-modal-footer">
          <button className="lyd-button ghost" onClick={onClose}>Close</button>
          <button className="lyd-button primary">Confirm</button>
        </div>
      </div>
    </div>
  );
};

Accessibility & Best Practices

Keyboard Navigation

  • Tab to navigate between modal elements
  • Escape to close modal
  • Enter/Space to activate buttons
  • Focus trap within modal content
  • Return focus to trigger element on close

Screen Reader Support

  • ARIA role="dialog" for modal containers
  • ARIA labelledby for modal titles
  • ARIA describedby for modal descriptions
  • ARIA hidden for background content
  • Live region announcements for state changes

Visual Accessibility

  • WCAG AA contrast ratios
  • Clear focus indicators
  • Readable typography
  • Reduced motion support

Next.js Integration

// components/Alert.tsx
import { useState } from 'react';

interface AlertProps {
  variant?: 'success' | 'error' | 'warning' | 'info';
  style?: 'default' | 'filled' | 'outlined';
  size?: 'small' | 'medium' | 'large';
  title: string;
  description?: string;
  icon?: React.ReactNode;
  actions?: { label: string; onClick: () => void }[];
  dismissible?: boolean;
  autoClose?: number; // in milliseconds
  onClose?: () => void;
  className?: string;
}

const defaultIcons = {
  success: '✓',
  error: '⚠',
  warning: '⚠',
  info: 'ℹ'
};

export const Alert = ({
  variant = 'info',
  style = 'default',
  size = 'medium',
  title,
  description,
  icon,
  actions = [],
  dismissible = true,
  autoClose,
  onClose,
  className = ''
}: AlertProps) => {
  const [isVisible, setIsVisible] = useState(true);

  const handleClose = () => {
    setIsVisible(false);
    onClose?.();
  };

  // Auto-close functionality
  useEffect(() => {
    if (autoClose && autoClose > 0) {
      const timer = setTimeout(handleClose, autoClose);
      return () => clearTimeout(timer);
    }
  }, [autoClose]);

  if (!isVisible) return null;

  const alertClasses = \`
    lyd-alert 
    \${variant} 
    \${style !== 'default' ? style : ''}
    \${size !== 'medium' ? size : ''}
    \${className}
  \`.trim();

  return (
    <div className={alertClasses} role="alert">
      {icon || defaultIcons[variant] && (
        <div className="lyd-alert-icon">
          {icon || defaultIcons[variant]}
        </div>
      )}
      
      <div className="lyd-alert-content">
        <div className="lyd-alert-title">{title}</div>
        {description && (
          <div className="lyd-alert-description">{description}</div>
        )}
        
        {actions.length > 0 && (
          <div className="lyd-alert-actions">
            {actions.map((action, index) => (
              <button
                key={index}
                className="lyd-alert-action"
                onClick={action.onClick}
              >
                {action.label}
              </button>
            ))}
          </div>
        )}
      </div>
      
      {dismissible && (
        <button className="lyd-alert-close" onClick={handleClose}>
          ×
        </button>
      )}
    </div>
  );
};

// Alert Context for Global Alerts
import { createContext, useContext, useState } from 'react';

interface AlertContextType {
  showAlert: (alert: Omit<AlertProps, 'onClose'>) => void;
  hideAlert: () => void;
}

const AlertContext = createContext<AlertContextType | undefined>(undefined);

export const AlertProvider = ({ children }: { children: React.ReactNode }) => {
  const [alert, setAlert] = useState<AlertProps | null>(null);

  const showAlert = (alertProps: Omit<AlertProps, 'onClose'>) => {
    setAlert({ ...alertProps, onClose: () => setAlert(null) });
  };

  const hideAlert = () => {
    setAlert(null);
  };

  return (
    <AlertContext.Provider value={{ showAlert, hideAlert }}>
      {children}
      {alert && <Alert {...alert} />}
    </AlertContext.Provider>
  );
};

export const useAlert = () => {
  const context = useContext(AlertContext);
  if (!context) {
    throw new Error('useAlert must be used within an AlertProvider');
  }
  return context;
};

// Usage Example:
export default function PropertyManagement() {
  const { showAlert } = useAlert();

  const handlePropertySold = () => {
    showAlert({
      variant: 'success',
      style: 'filled',
      title: 'Property Sold Successfully!',
      description: 'Congratulations! Your property has been sold. The buyer will contact you for next steps.',
      actions: [
        { label: 'View Details', onClick: () => console.log('View details') },
        { label: 'Contact Buyer', onClick: () => console.log('Contact buyer') }
      ],
      autoClose: 5000
    });
  };

  const handleInquiryError = () => {
    showAlert({
      variant: 'error',
      title: 'Failed to Send Inquiry',
      description: 'There was an error sending your property inquiry. Please check your connection and try again.',
      actions: [
        { label: 'Retry', onClick: () => console.log('Retry') },
        { label: 'Contact Support', onClick: () => console.log('Support') }
      ]
    });
  };

  return (
    <div>
      <button onClick={handlePropertySold}>Mark as Sold</button>
      <button onClick={handleInquiryError}>Test Error</button>
      
      <Alert
        variant="warning"
        title="Price Over Budget"
        description="This property exceeds your specified budget by €150,000."
        actions={[
          { label: 'Adjust Budget', onClick: () => console.log('Adjust budget') },
          { label: 'View Similar', onClick: () => console.log('View similar') }
        ]}
      />
    </div>
  );
}

Accessibility & Best Practices

ARIA Support

Proper role="alert" and live regions for screen readers to announce important messages immediately upon appearance.

Keyboard Navigation

Full keyboard support for dismissible alerts and action buttons, ensuring all interactive elements are accessible via Tab navigation.

Screen Reader Support

ARIA labels and semantic HTML ensure compatibility with assistive technologies.

Color Independence

Icons and text content provide meaning beyond color, ensuring accessibility for users with color vision deficiencies.

Focus Management

Proper focus handling when alerts appear and disappear, maintaining logical tab order and user orientation.

Content Clarity

Clear, actionable messaging with descriptive titles and specific instructions for user actions.

Auto-Dismiss Options

Configurable auto-dismiss timing that respects user reading speed and provides manual dismiss controls.

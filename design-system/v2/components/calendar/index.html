React/Next.js Component

// components/Calendar.tsx
import { useState } from 'react';

interface Event {
  id: string;
  date: Date;
  time: string;
  title: string;
  type: 'viewing' | 'meeting' | 'deadline';
}

interface CalendarProps {
  events?: Event[];
  selectedDate?: Date;
  onDateSelect?: (date: Date) => void;
}

export const Calendar = ({
  events = [],
  selectedDate = new Date(),
  onDateSelect
}: CalendarProps) => {
  const [currentMonth, setCurrentMonth] = useState(selectedDate);
  const [view, setView] = useState<'month' | 'year' | 'decade'>('month');
  
  const getDaysInMonth = (date: Date) => {
    const year = date.getFullYear();
    const month = date.getMonth();
    const firstDay = new Date(year, month, 1);
    const lastDay = new Date(year, month + 1, 0);
    const days = [];
    
    // Previous month days
    const startDay = firstDay.getDay();
    for (let i = startDay - 1; i >= 0; i--) {
      const day = new Date(year, month, -i);
      days.push({ date: day, otherMonth: true });
    }
    
    // Current month days
    for (let i = 1; i <= lastDay.getDate(); i++) {
      const day = new Date(year, month, i);
      days.push({ date: day, otherMonth: false });
    }
    
    // Next month days
    const endDay = lastDay.getDay();
    for (let i = 1; i < 7 - endDay; i++) {
      const day = new Date(year, month + 1, i);
      days.push({ date: day, otherMonth: true });
    }
    
    return days;
  };
  
  const hasEvent = (date: Date) => {
    return events.some(event => 
      event.date.toDateString() === date.toDateString()
    );
  };
  
  const isToday = (date: Date) => {
    return date.toDateString() === new Date().toDateString();
  };
  
  const isSelected = (date: Date) => {
    return date.toDateString() === selectedDate.toDateString();
  };
  
  return (
    <div className="lyd-calendar">
      <div className="lyd-calendar-header">
        <div className="lyd-calendar-title">
          {currentMonth.toLocaleDateString('en-US', { 
            month: 'long', 
            year: 'numeric' 
          })}
        </div>
        <div className="lyd-calendar-nav">
          <button onClick={() => setCurrentMonth(/* prev month */)}>
            <svg>...</svg>
          </button>
          <button onClick={() => setCurrentMonth(/* next month */)}>
            <svg>...</svg>
          </button>
        </div>
      </div>
      
      {/* Calendar grid */}
      <div className="lyd-calendar-grid">
        {['Su', 'Mo', 'Tu', 'We', 'Th', 'Fr', 'Sa'].map(day => (
          <div key={day} className="lyd-calendar-weekday">{day}</div>
        ))}
        
        {getDaysInMonth(currentMonth).map(({ date, otherMonth }) => (
          <button
            key={date.toISOString()}
            className={`
              lyd-calendar-day
              ${otherMonth ? 'other-month' : ''}
              ${isToday(date) ? 'today' : ''}
              ${isSelected(date) ? 'selected' : ''}
              ${hasEvent(date) ? 'has-event' : ''}
            `}
            onClick={() => onDateSelect?.(date)}
          >
            {date.getDate()}
          </button>
        ))}
      </div>
    </div>
  );
};

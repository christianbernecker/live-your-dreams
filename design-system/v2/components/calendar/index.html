Next.js Integration

import { useState, useEffect } from 'react';

interface CalendarProps {
  mode: 'single' | 'range';
  onDateSelect?: (date: Date) => void;
  onRangeSelect?: (startDate: Date, endDate: Date) => void;
  className?: string;
}

export const LydCalendar: React.FC<CalendarProps> = ({
  mode = 'single',
  onDateSelect,
  onRangeSelect,
  className = ''
}) => {
  const [currentDate, setCurrentDate] = useState(new Date());
  const [selectedDate, setSelectedDate] = useState<Date | null>(null);
  const [rangeStart, setRangeStart] = useState<Date | null>(null);
  const [rangeEnd, setRangeEnd] = useState<Date | null>(null);
  const [days, setDays] = useState<Array<{
    date: Date;
    day: number;
    isOtherMonth: boolean;
    isToday: boolean;
  }>>([]);

  const monthNames = [
    'January', 'February', 'March', 'April', 'May', 'June',
    'July', 'August', 'September', 'October', 'November', 'December'
  ];

  useEffect(() => {
    generateCalendarDays();
  }, [currentDate]);

  const generateCalendarDays = () => {
    // Calendar generation logic
    const year = currentDate.getFullYear();
    const month = currentDate.getMonth();
    const today = new Date();
    
    // Generate days array with proper date calculations
    const calendarDays = [];
    // ... implementation details
    
    setDays(calendarDays);
  };

  const handleDateClick = (date: Date) => {
    if (mode === 'single') {
      setSelectedDate(date);
      onDateSelect?.(date);
    } else {
      // Range selection logic
      if (!rangeStart || (rangeStart && rangeEnd)) {
        setRangeStart(date);
        setRangeEnd(null);
      } else {
        const endDate = date < rangeStart ? rangeStart : date;
        const startDate = date < rangeStart ? date : rangeStart;
        setRangeStart(startDate);
        setRangeEnd(endDate);
        onRangeSelect?.(startDate, endDate);
      }
    }
  };

  return (
    <div className={`lyd-calendar ${mode === 'range' ? 'range-mode' : ''} ${className}`}>
      {mode === 'range' && (
        <div className="lyd-calendar-range-display">
          <span className="date-range">
            {rangeStart && rangeEnd 
              ? `${rangeStart.toLocaleDateString()} - ${rangeEnd.toLocaleDateString()}`
              : 'Select start and end date'
            }
          </span>
        </div>
      )}
      
      <div className="lyd-calendar-header">
        <button 
          className="lyd-calendar-nav prev-btn"
          onClick={() => setCurrentDate(new Date(currentDate.getFullYear(), currentDate.getMonth() - 1))}
        >‹</button>
        <h3 className="lyd-calendar-title">
          {monthNames[currentDate.getMonth()]} {currentDate.getFullYear()}
        </h3>
        <button 
          className="lyd-calendar-nav next-btn"
          onClick={() => setCurrentDate(new Date(currentDate.getFullYear(), currentDate.getMonth() + 1))}
        >›</button>
      </div>
      
      <div className="lyd-calendar-grid">
        {['Mo', 'Tu', 'We', 'Th', 'Fr', 'Sa', 'Su'].map(day => (
          <div key={day} className="lyd-calendar-day-header">{day}</div>
        ))}
        
        {days.map((dayData, index) => (
          <div
            key={index}
            className={`lyd-calendar-day ${dayData.isOtherMonth ? 'other-month' : ''} ${dayData.isToday ? 'today' : ''}`}
            onClick={() => handleDateClick(dayData.date)}
          >
            {dayData.day}
          </div>
        ))}
      </div>
    </div>
  );
};

// Usage Examples

  const handleDateSelect = (date: Date) => {

  };

    console.log('Property available from:', startDate, 'to:', endDate);
  };

  return (
    <div>
      <h3>Property Viewing Date</h3>

    </div>
  );
}
